import numpy as np
import pandas as pd
from scipy.stats import t

x = [3.5, 2.5, 4, 3.8, 2.8, 1.9, 3.2, 3.7, 2.7, 3.3]
y = [3.3, 2.2, 3.5, 2.7, 3.5, 2, 3.1, 3.4, 1.9, 3.7]

# 편차
x_d = x - np.mean(x)
y_d = y - np.mean(y)
xy_d = x_d * y_d

# 기울기 b1
slope = sum(xy_d)/sum(x_d**2)

# 절편 b0
intercept = np.mean(y) - slope * np.mean(x)
